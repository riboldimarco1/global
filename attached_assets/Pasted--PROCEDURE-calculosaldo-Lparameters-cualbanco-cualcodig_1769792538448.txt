
PROCEDURE calculosaldo

Lparameters cualbanco,cualcodigo,cambiofecha
Local thislock
If lastwindow<>"*"
	If Vartype(&lastwindow)="O".And.Type("&lastwindow..lockscreen")<>"U"
		thislock=&lastwindow..LockScreen
		&lastwindow..LockScreen=.T.
	Endif
Endif
Do salvaset
Local tempmessage,cualfiltro,ttt,calcorder,calcfilter
If .Not.Empty(cualbanco)
	Select bancos
	Set Filter To banco=cualbanco
	Count To ttt
	If ttt=0
	Else
		Set Order To FECHA
		Set Deleted Off
		Seek cualcodigo Order codigoauto
		If .Not. Found().Or.cambiofecha
			cambiofecha=.F.
			Go Top
		Endif
		thisfecha=FECHA
		Set Deleted On
		If .Not. Bof()
			Skip -1
		Endif
		If.Not.Bof()
			tempsaldo=saldo
			tempbansaldo=saldoconci
		Else
			If tipoop="suma"
				tempsaldo=monto
			Else
				tempsaldo=-monto
			Endif
			If conciliado
				tempbansaldo=monto
			Else
				tempbansaldo=0
			Endif
			Replace saldo With tempsaldo,saldoconci With tempbansaldo
		Endif
		tempcod=codigoauto
		tempfecha=FECHA
		If ttt<>1
			Skip
		Endif
		If .Not."euro"$cualbanco.And..Not."dolares"$cualbanco
			If tempfecha<Ctod("18/08/2018").And.FECHA=>Ctod("18/08/2018")
				tempsaldo=tempsaldo/100000
				tempbansaldo=tempbansaldo/100000
			Endif
		Endif
		Do While.Not.Eof()
			If tipoop="suma"
				If ttt<>1
					tempsaldo=tempsaldo+monto
				Else
					tempsaldo=monto
				Endif
				If conciliado
					If ttt<>1
						tempbansaldo=tempbansaldo+monto
					Else
						tempbansaldo=monto
					Endif
				Endif
			Else
				If ttt<>1
					tempsaldo=tempsaldo-monto
				Else
					tempsaldo=-monto
				Endif
				If conciliado
					If ttt<>1
						tempbansaldo=tempbansaldo-monto
					Else
						tempbansaldo=-monto
					Endif
				Endif
			Endif
			Replace saldo With tempsaldo,saldoconci With tempbansaldo
			tempfecha=FECHA
			Skip
			If .Not."euro"$cualbanco.And..Not."dolares"$cualbanco
				If tempfecha<Ctod("18/08/2018").And.FECHA=>Ctod("18/08/2018")
					tempsaldo=tempsaldo/100000
					tempbansaldo=tempbansaldo/100000
				Endif
			Endif
		Enddo
		Flush
	Endif
ENDIF
flush
Do restauraset
If lastwindow<>"*"
	If Vartype(&lastwindow)="O".And.Type("&lastwindow..lockscreen")<>"U"
		&lastwindow..LockScreen=thislock
	Endif
Endif

If Type("windowsaldo")="O"
	windowsaldo.timer1.Enabled=.T.
Endif




ENDPROC
